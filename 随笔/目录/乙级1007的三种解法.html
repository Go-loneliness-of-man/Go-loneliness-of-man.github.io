PAT 乙级 1007 的三种解法

2018 年 8 月 29 日

共四种算法，前两种求素数的方法类似，算一种，后两种运用了经典算法。运算时间以 PAT 网站测试样例 5 的运算时间为衡量标准。
注意：求素数时，不必取余到相等，取余到根号即可，具体代码是：< (int)sqrt(n) + 1

一、打表

即求出 n 以下所有素数存到数组中，然后求差值计数，这种方法简称打表。样例 5 运行时间是 14 ms #include #include //d = y - x，y 是 x 的下一个素数，二者是相邻素数，现在假设有无数对 x、y，且其对应的 d 为 2，求指定范围内有多少对这样的 x、y int main() { int n, nums[30000] = { 0 }, i, j, k, temp, x; //范围 n、素数列 nums、循环变量 i 和 j、计算满足条件的 k、临时变量 temp、数组编号 x scanf_s("%d", &n); nums[0] = 2; for (k = x = 0, i = 3; i <= n; i++) //求出 n 内所有素数 { for (j = 2, temp = (int)sqrt(i) + 1; j < temp; j++) //判断 i 是否是素数 if (i%j == 0) break; //不是素数，跳出 if (temp != j) continue; //未循环到相等便跳出，不是素数，跳过此次循环 nums[x++] = i; } for (i = 0; nums[i] != 0; i++) if (nums[i + 1] - nums[i] == 2) k++; //二者是相邻且差值为 2 的素数对，满足条件 printf("%d", k); return 0; } /* 思路： 将 n 内所有素数求出存储到数组中，循环计算差值计数 */

二、利用两个变量交替计算差值

不用数组，循环计算素数，每次找出素数存到变量中进行差值运算，计数，这种方法不用预估素数的个数，能省去数组占用的无用存储空间。 样例 5 的运算时间是 14 ms #include<stdio.h> //d = y - x，y 是 x 的下一个素数，二者是相邻素数，现在假设有无数对 x、y，且其对应的 d 为 2，求指定范围内有多少对这样的 x、y int main() { int n, x, y, i, j, k; //范围、素数对 xy、循环变量 i 和 j、计算满足条件的 k、临时变量 temp scanf("%d", &n); for (k = 0, x = 2, y = 3, i = 3; i <= n; i++) //求出 n 内所有素数 { for (j = 2; j < i; j++) //判断 i 是否是素数 if (i%j == 0) break; //不是素数，跳出 if (i != j) continue; //未循环到相等便跳出，不是素数，跳过此次循环 if (i != 3) //对 x、y 的值进行更新 { x = y; //更新 x y = i; //更新 y } if (y - x == 2) k++; //二者是相邻且差值为 2 的素数对，满足条件 } printf("%d", k); return 0; } /* 思路： x、y 初始设置为 1、3，利用循环筛选素数，对 x、y 进行更新，差值为 2 则计数变量自增一次 */

三、利用埃氏筛大幅减少运算量

埃氏筛：求出根号 n 范围下所有素数形成数列 x，然后，依次将 n 内的，是数列 x 中某数的整数倍的数字去掉，剩下的就是素数。样例 5 运行时间 6 ms
下面是两种实现方法，第一个分步写，结构清晰，易于理解，代码量比较大。另一个一步到位，结构紧凑，较难理解，代码量少。两种实现方法时间复杂度都差不多。 #include<stdio.h> #include<math.h> //d = y - x，y 是 x 的下一个素数，二者是相邻素数，现在假设有无数对 x、y，且其对应的 d 为 2，求指定范围内有多少对这样的 x、y int main() { int n, nums[30000] = { 0 }, nums2[300] = { 0 }, i, j, k, temp, x; //范围 n、素数列 nums、根号素数列 nums2、循环变量 i 和 j、计算满足条件的 k、临时变量 temp、数组编号 x scanf("%d", &n); nums2[0] = 2; for (x = 1, i = 3; i < (int)sqrt(n) + 1; i++) //求出根号素数列 { for (j = 2, temp = (int)sqrt(i) + 1; j < temp; j++) //判断 i 是不是素数，利用求根号大幅减少了运算量 if (i%j == 0) break; //整除，不是素数，跳出 if (temp != j) continue; //不等于，循环提前结束，不是素数，跳过 nums2[x++] = i; //赋值根号素数列 } for (i = 0; nums2[i] != 0; i++) nums[i] = nums2[i]; //拷贝 nums2 到 nums 中 for (i = (int)sqrt(n) + 1; i <= n; i++) //循环赋值 nums { for (j = 0; nums2[j] != 0; j++) //判断 i 是否是 nums2 任一元素的倍数 if (i%nums2[j] == 0) break; //不是素数，跳出 if (nums2[j] != 0) continue; //未循环完，不是素数，跳过 nums[x++] = i; //将筛选出的素数赋值到 nums 中 } for (i = k = 0; nums[i + 1] != 0; i++) if (nums[i + 1] - nums[i] == 2) k++; printf("%d", k); return 0; } /* 思路： 利用埃氏筛快速找出 n 以内所有素数并存到数组中，对数组的相邻元素比较，差值为 2 则 k 自增一次。 埃氏筛实现：将根号 n 下所有素数求出存到 nums2 数组中，接下来就是去掉 n 内所有 nums2 中元素的倍数。去掉这个操作可用 if 判断，即使循环变量 i 自增 赋值到 nums 中，赋值前用 if 判断 i 是否能整除 nums2 中的任一元素（取余为 0），若是则跳过该次循环不赋值，便实现了去掉该数，将这个 if 放入 for 中 循环判断是否是 nums2 中任一元素的倍数，若没有则为素数，赋值。即两层 for，外层提供 i，内层筛选 i 是否满足条件。 注意： 1. 为 nums 赋值时 i 的初始值直接为根号 n。 2. nums 的前面一部分直接拷贝 nums2。 3. 求素数时，不必循环取余到 i，只需取余到 (int)sqrt(n) + 1 便可以判定是否为素数，这可大幅减少运算量。 */ #include<stdio.h> #include<math.h> //d = y - x，y 是 x 的下一个素数，二者是相邻素数，现在假设有无数对 x、y，且其对应的 d 为 2，求指定范围内有多少对这样的 x、y int main() { int n, nums[30000] = { 0 }, i, j, k, temp, x; //范围 n、素数列 nums、循环变量 i 和 j、计算满足条件的 k、临时变量 temp、数组编号 x scanf_s("%d", &n); nums[0] = 2; for (x = 0, i = 3; i <= n; i++) //求出 n 内所有素数 { for (j = 0, k = 1, temp = (int)sqrt(i) + 1; nums[j] < temp; j++) //判断 i 是否是素数 if (i%nums[j] == 0) { k = 0; //标记状态，i 不是素数 break; //不是素数，跳出 } if (k == 0) continue; //未循环到数组末尾便跳出或到末尾，不是素数，跳过此次循环 nums[++x] = i; //赋值素数到 nums } for (i = k = 0; nums[i] != 0; i++) if (nums[i + 1] - nums[i] == 2) k++; //二者是相邻且差值为 2 的素数对，满足条件 printf("%d", k); return 0; } /* 思路： 利用埃氏筛法快速找出 n 以内所有素数并存到数组中，对数组的相邻元素比较，差值为 2 则 k 自增一次。 埃氏筛法实现：for 中先为 nums[0] 赋初值为 2，循环变量 i 从 3 一直自增到大于 n。循环中判断 i 是否是素数，若是则赋值到 nums，否则 不赋值。判断方法是内嵌一个 for 遍历 nums，只要 nums[j] < (int)sqrt(i) + 1 便不终止，过程中用 i % nums[j] 判断是否能整除， 若能则不是素数，标记变量 k 为 0，k 为 0 则跳过该 i 不赋值到 nums 中。 注意： 1. 求素数时，不必循环取余到 i，只需取余到 (int)sqrt(n) + 1 便可以判定是否为素数，这可大幅减少运算量。 2. 注意循环初值。 */

四、利用欧拉筛法减少更多运算量

埃氏筛在过程中还是有很多重复，例如重复去掉 2、3 的共同倍数，而欧拉筛可彻底避免重复，欧拉筛法的思想是确保每个数只被其最小的质因数筛去一次。
